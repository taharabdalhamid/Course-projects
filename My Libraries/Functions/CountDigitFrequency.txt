int CountDigitFrequency(short DigitToCheck, int Number)
{

    int FreqCount = 0;
    int Remainder = 0;

    while (Number > 0)
    {
        Remainder = Number % 10;
        Number = Number / 10;

        if (Remainder == DigitToCheck)
        {
            FreqCount++;
        }

    }

    return FreqCount;
}
